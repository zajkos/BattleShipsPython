-- 1. Tworzenie tabeli Użytkowników
CREATE TABLE users (
    user_id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 2. Tworzenie tabeli Meczów
CREATE TABLE matches (
    match_id SERIAL PRIMARY KEY,
    player1_id INT NOT NULL,
    player2_id INT NOT NULL,
    winner_id INT,
    played_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT fk_player1 FOREIGN KEY(player1_id) REFERENCES users(user_id) ON DELETE CASCADE,
    CONSTRAINT fk_player2 FOREIGN KEY(player2_id) REFERENCES users(user_id) ON DELETE CASCADE,
    CONSTRAINT fk_winner FOREIGN KEY(winner_id) REFERENCES users(user_id) ON DELETE SET NULL
);

-- 3. Tworzenie tabeli Wyników
CREATE TABLE scores (
    score_id SERIAL PRIMARY KEY,
    user_id INT NOT NULL,
    match_id INT NOT NULL,
    points INT NOT NULL,
    is_winner BOOLEAN NOT NULL,
    CONSTRAINT fk_user FOREIGN KEY(user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    CONSTRAINT fk_match FOREIGN KEY(match_id) REFERENCES matches(match_id) ON DELETE CASCADE
);
